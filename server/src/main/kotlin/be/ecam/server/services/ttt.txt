// services/admin_service.kt : service for admin management (CRUD)

package be.ecam.server.services

import at.favre.lib.crypto.bcrypt.BCrypt
import be.ecam.server.models.*
import be.ecam.server.models.UpdateAdminRequest as updateAdminRequest
import org.jetbrains.exposed.sql.SqlExpressionBuilder.eq
import org.jetbrains.exposed.sql.transactions.transaction


object AdminService {

    

    // read all admins
    fun getAllAdmins(): List<AdminDTO> = transaction {
        Admin.all().map { a ->
            AdminDTO(
                id = a.id.value,
                username = a.username,
                email = a.email
            )
        }

    }

    // read admin by id
    fun getAdminById(id: Int): AdminDTO? = transaction {
        val a = Admin.findById(id) ?: return@transaction null
        AdminDTO(
            id = a.id.value,
            username = a.username,
            email = a.email
        )
    }

    // update admin
    fun updateAdmin(id: Int, req: updateAdminRequest): AdminDTO? = transaction {
        val a = Admin.findById(id) ?: return@transaction null

        req.username?.let { a.username = it.trim() }
        req.email?.let { a.email = it.trim() }
        req.password?.let {
            val hashed = BCrypt.withDefaults().hashToString(12, it.toCharArray())
            a.password = hashed
        }
        AdminDTO(
            id = a.id.value,
            username = a.username,
            email = a.email
        )
    }

    // delete admin
    fun deleteAdmin(id: Int): Boolean = transaction {
        val a = Admin.findById(id) ?: return@transaction false
        a.delete()
        true
    }

    

}
